find_package(ADTF REQUIRED PATHS "${ADTF_DIR}")

cmake_policy(PUSH)
cmake_policy(SET CMP0023 OLD)

adtf_add_filter(convoy_track_extractor_filter
    #>>> src/*.h
    src/convoy_track_extractor_filter.h
    #>>> src/*.cpp
    src/convoy_track_extractor_filter.cpp
)

target_link_libraries(convoy_track_extractor_filter
    PRIVATE
        adtf_plugin_lib
        positioning_server
        convoy_tracking
        math_library
)

set_property(TARGET convoy_track_extractor_filter PROPERTY FOLDER road_fusion_adtf)

eb_target_add_logging(convoy_track_extractor_filter
    CONVOY_TRACK_EXTRACTOR_FILTER
)

adtf_install_plugin(convoy_track_extractor_filter ./bin)

cmake_policy(POP)
